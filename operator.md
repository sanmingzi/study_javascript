# 运算符

## 数学运算

```
+-*/%
```

```
var x = 1, y = 1;
x = y++ // x = 1, y = 2
x = ++y // x = 3, y = 3
var z = "1"
x + z // "31",数字和字符串相加结果为数字
```

```
&& || !
```

## 位运算

- 按位与 &

```
1 & 2 // 01 & 10 => 0
1 & 3 // 01 & 11 => 1
if(n & 1)
{
  console.log("n是基数")
}
```

- 按位或 |

```
1 || 2 // 01 | 10 => 3
1 || 4 // 01 | 100 => 5

// 浮点型向下取整
var num = 1.1 | 0
```

- 按位非 ~

```
var num = 1
var num1 = ~num
//  num: 00000000000000000000000000001
// ~num: 11111111111111111111111111110
// js中默认数字是有符号的，所以取非之后的数字为负数
// 负数二进制转化为十进制：最高位符号位不变，其他位取反，然后加1
// 1000000000000000000000010 => -2
```

- 有符号左移 <<

```
// 这个其实相当于原来的数字 *2
// 原理大概就是二进制的0和1是不变的，但是都升高了1位，所以恰好是 *2
3 << 1 // 6
3 << 2 // 12

// 计算2的n次方
function power(n) {
  return 1 << n;
}
```

- 有符号右移 >>

```
// 求一个数的二分之一
var num = 64 >> 1; // 32
var num1 = 64 >> 5; // 2
```

- 无符号右移 >>>

```
// 无符号右移会影响符号位，负数会变成非负数
var num = -1; // 二进制：1111...111
num = num >>> 0; // 二进制：0111...111，十进制：4294967295

// 判断一个数为非负数
function isPos(num) {
  return (num >>> 0) == num
}
```
